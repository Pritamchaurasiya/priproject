from typing import List def max_subarray_sum(arr: List[int]) -> int: """ You are given an array of integers. Your task is to find the maximum sum of a contiguous subarray within the array. The subarray must contain at least one element. As a reminder, your code has to be in python """ if not arr: return 0 max_current = max_global = arr[0] # Bolt: Optimization - Use iterator to avoid indexing overhead and replace max() # with conditional check to avoid function call overhead. it = iter(arr) next(it) for x in it: if max_current < 0: max_current = x else: max_current += x if max_current > max_global: max_global = max_current return max_global
